constexpr:

编译器计算常量, 甚至可以修饰函数（接受 const 变量）

constexpr int n=3+2+1;
constexpr int fib(const int n){
    // c++11, 如果使用if而不是三元运算符需要c++14
    return n==1 || n==2 ? 1 : fib(n-1)+fib(n-2);
}

int array[fib(10) + 5];

例：

#include <iostream>
#include <array>
using namespace std;

constexpr int n=3+2+1;
constexpr int fib(const int n){
    // c++11, 如果使用if而不是三元运算符需要c++14
    return n==1 || n==2 ? 1 : fib(n-1)+fib(n-2);
}

std::array<int, fib(10) + 5> Array;
int main() {
    cout<<Array.size()<<endl;
    return 0;
}

nullptr:

NULL 会随着编译器的不同被解释成((void*)0) 和0, c++不允许将void*指针隐式转换成别的类型

于是char *ch=NULL之中只好定义 NULL 为 0

但这又会出现重载的反直觉行为

int foo(char*){
    // 1
}
int foo(int){
    // 2
}
foo(NULL); // 等效于foo(0)，调用2而不是1

为了解决这个问题，c++引入了 nullptr， 并且定义它的类型是 nullptr_t，可以隐式转换成任意指针和成员指针的类型，并做比较

#include <initializer_list>
class Magic{
private:
    vector<int> numbers;
public:
     Magic(std::initializer_list ilist){
        for(auto i: ilist){

        }
    }

Magic m={1,2,3,4,5}; // act as args in c
// 也可以用initializer_list做普通函数的形参
void foo(std::initializer_list<int> ilist){
    //...
}

decltype

为了解决 auto 只能推断 变量 的不足

引入了 decltype 来推断表达式等，可以和std::is_same配合

if(std::is_same<decltype(x+y*z), int>::value)
    std::cout<<"is int"<<endl;

尾返回类型

// 传统c++
template<typename R, typename T, typename U>
R add(T x, U y) {
    return x+y
}
// 不足：1.丑 2.需要指定R，但实际上不知道R是什么类型

// modern cpp
template<typename T, typename U>
auto add(T x, U y)->decltype(x+y){
    return x+y;
}
// 甚至，c++14
template<typename T, typename U>
auto add(T x, U y){
    return x+y;
}

区间 for

#include <iostream>
#include <vector>
#include <algorithm>
int main() {
    std::vector<int> vec = {1, 2, 3, 4};
    if (auto itr = std::find(vec.begin(), vec.end(), 3); itr != vec.end()) *itr = 4;
    for (auto element : vec)
        std::cout << element << std::endl; // read only
    for (auto &element : vec) {
        element += 1;                      // writeable
    }
}

类型别名模板

typedef 只能作用于类型

但是函数指针等特殊类型，模板（不是类型！）等不可以 typedef

使用 using 语法代替

using NewProcess=int (*)(void*);
auto p = new NewProcess();

class foo{
public:
    T type;

    // ...

template<typename T>
using somefoo=foo<std::vector<T>, std::string>;  // somefoo is a template now
somefoo<int> s;

变长模板

著名黑魔法之一

// sizeof和打包
template<typename... Ts>
void magic(Ts... args){
    cout<<sizeof...(args)<<endl;
}

// 解包1: 递归
template<typename T0>
void myprintf(T0 x){
    cout<<x<<endl;
}
template<typename T, typename ...Ts>
void myprintf(T value, Ts ...args){
    cout<<value<<endl;
    myprintf(args...);
}

myprintf(1, 0.2, "0.1", "haha");

// 解包2: 变参数模板展开 c++17
template<typename T0, typename... Ts>
void myprintf(T0 t, Ts ...args){
    cout<<t<<endl;
    if constexpr(sizeof...(args) > 0) myprintf(args...);
}

线程和锁

#include<thread>
void critical_section(int v){
    static std::mutex mtx;
    std::unique_lock<std::mutex> lock;
    lock.lock();
    // 竞争代码
    v += 1

    lock.unlock();
}
int main(){
    std::thread t1(critical_section, 10),t2(critical_section, 20);
    t1.join();
    t2.join();
}
